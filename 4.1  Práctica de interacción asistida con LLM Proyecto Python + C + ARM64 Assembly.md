# 🧠 PROYECTO: Integración Python + C + ARM64 Assembly  
## 🔥 PyCAsm Performance Lab

Este proyecto demuestra la interacción de bajo nivel entre **Python, C y ARM64 Assembly**, analizando el rendimiento y la comunicación entre lenguajes mediante librerías compartidas (`.so`) y la interfaz `ctypes`.

    // Autor: Javier Saenz Gallegos
    // Matrícula: 22210352
    // Fecha: 2026-04-16
    // Descripción: Funciones matemáticas en ARM64 (AArch64)

    .global suma_asm
    .global resta_asm
    .global max_asm

    .text
    .align 2

    suma_asm:
        add x0, x0, x1
        ret

    resta_asm:
        sub x0, x0, x1
        ret

    max_asm:
        cmp x0, x1
        csel x0, x0, x1, gt
        ret

### 🐍 Python Interface (`src/main.py`)

    # Autor: Javier Saenz Gallegos
    # Matrícula: 22210352
    # Descripción: Interfaz Python para librería C/ASM

    import ctypes
    import os

    lib_path = os.path.abspath('./build/libcalc.so')
    lib = ctypes.CDLL(lib_path)

    lib.suma.argtypes = [ctypes.c_int, ctypes.c_int]
    lib.suma.restype = ctypes.c_int

    def test():
        print("--- PyCAsm Performance Lab ---")
        res = lib.suma(5, 3)
        print(f"Resultado (Python -> C -> ASM): {res}")

    if __name__ == "__main__":
        test()

## 🏗️ Compilación y Ejecución

### Makefile

    CC=gcc
    CFLAGS=-fPIC -shared -g

    all:
    	@mkdir -p build
    	$(CC) $(CFLAGS) src/c_lib.c src/asm_lib.s -o build/libcalc.so
    	@echo "Librería compilada en build/libcalc.so"

    clean:
    	rm -rf build/

### Ejecutar

    make
    python3 src/main.py

## ⏱️ Benchmark (`tests/benchmark.py`)

    import time
    import ctypes
    import os

    lib = ctypes.CDLL(os.path.abspath('../build/libcalc.so'))

    def benchmark():
        N = 1000000
        print(f"Ejecutando {N} iteraciones...")
        
        start = time.time()
        for _ in range(N):
            lib.suma(5, 3)
        end = time.time()

        print(f"Tiempo Total C/ASM: {end - start:.4f} segundos")

    if __name__ == "__main__":
        benchmark()
